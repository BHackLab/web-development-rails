ruby

structures de contrôle - elsif


print "Would you like to see your name reverse, uppercase or both? (reverse, uppercase both or no)"
answer = gets.chomp.downcase

if answer == "reverse"
  puts "This is your name backwards:"
  puts name.reverse
elsif answer == "uppercase"
  puts "This is your name in all uppercase letters:"
  puts name.upcase
elsif answer == "both"
  puts name.reverse.upcase
else
  puts "Ok, maybe later."
end

conditions logique - OR et AND


print "What is your favorite number? "
number = gets.chomp.to_i

if (number == 3) || (number == 5)
  puts "That's my favorite number!"
elsif (number > 10) && (number.even?)
  puts "That's a pretty high even number!"
elsif (number.odd?) && (number % 3 == 0)
  puts "That number is divisible by 3 and odd, cool!"
end
