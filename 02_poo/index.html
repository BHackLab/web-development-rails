Écrire une classe Name qui retourne, le prénom, le nom, le nom complet


class Name
  def first_name
    "Benjamin"
  end

  def last_name
    "Guimberteau"
  end

  def full_name
    "#{first_name} #{last_name}"
  end
end

name = Name.new
puts name.first_name
puts name.last_name
puts name.full_name

bases - Namespace

Ajouter un fichier laser_bots.rb


module LaserBots
  module Console
    class Command
    end
  end

  module World
    class Player
      attr_reader :name
      def initialize(name)
        @name = name
      end
    end
  end
end

irb> load "./laser_bots.rb"
=> true
            

player = LaserBots::World::Player.new("Jason")
puts player.name
